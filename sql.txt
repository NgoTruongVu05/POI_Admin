CREATE DATABASE IF NOT EXISTS POI_Database CHARACTER SET utf8mb4 COLLATE utf8mb4_unicode_ci;
USE POI_Database;

CREATE TABLE IF NOT EXISTS categories (
    id VARCHAR(20) PRIMARY KEY,
    name NVARCHAR(100) NOT NULL,     
    color VARCHAR(20) DEFAULT 'blue' -- Lưu tên màu Tailwind: blue, orange, green, red...
) ENGINE=InnoDB;

CREATE TABLE IF NOT EXISTS users (
    uid VARCHAR(50) PRIMARY KEY,
    username VARCHAR(100) UNIQUE NOT NULL,
    password VARCHAR(255) NOT NULL
) ENGINE=InnoDB;

CREATE TABLE IF NOT EXISTS pois (
    id VARCHAR(50) PRIMARY KEY,
    name VARCHAR(200) NOT NULL,
    description TEXT,
    categoryId VARCHAR(20),
    lat DOUBLE NOT NULL,
    lng DOUBLE NOT NULL,
    CONSTRAINT fk_category FOREIGN KEY (categoryId)
        REFERENCES categories(id) ON DELETE SET NULL ON UPDATE CASCADE
) ENGINE=InnoDB;

CREATE TABLE IF NOT EXISTS tours (
    id VARCHAR(50) PRIMARY KEY,
    name VARCHAR(200) NOT NULL,
    description TEXT
) ENGINE=InnoDB;

CREATE TABLE IF NOT EXISTS tour_details (
    tourId VARCHAR(50),
    poiId VARCHAR(50),
    sortOrder INT, -- Thứ tự di chuyển trong tour
    PRIMARY KEY (tourId, poiId),
    CONSTRAINT fk_tour FOREIGN KEY (tourId)
        REFERENCES tours(id) ON DELETE CASCADE,
    CONSTRAINT fk_poi FOREIGN KEY (poiId)
        REFERENCES pois(id) ON DELETE CASCADE
) ENGINE=InnoDB;
